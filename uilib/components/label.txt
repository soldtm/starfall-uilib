--@name UILib Label
--@author soldnotsold
--@client

if not UILib then
    throw("Import uilib/base first!")
end

local table = table
local math = math
local render = render

--[[
  _         _         _ 
 | |   __ _| |__  ___| |
 | |__/ _` | '_ \/ -_) |
 |____\__,_|_.__/\___|_|
--]]
---@class UILib.Label: UILib.Component
---@field private font any
---@field protected textColor Color
---@field icon UILib.Icon | nil
---@field iconSize integer
---@field iconColor Color | nil
---@field iconPosition string
---@field iconPadding integer
UILib.Label = class("Label", UILib.Component)

function UILib.Label:initialize(parent, screen)
    self.font = UILib.Skin.Label.font
    self.textColor = UILib.Skin.Label.textColor

    UILib.Component.initialize(self, 0,0, 0, 0, false, parent, screen)
    self.icon = nil
    self.iconSize = 16
    self.iconColor = nil
    self.iconPosition = "left"
    self.iconPadding = 4

    self:setText("Label")
end

function UILib.Label:setFont(font)
    self.font = font
    self:setText(self.text)
end

function UILib.Label:setColor(clr)
    self.textColor = clr
end

---Sets position of an icon
---@param position string "left", "right", "top", "bottom"
function UILib.Label:setIconPosition(position)
    local validPositions = {"left", "right", "top", "bottom"}
    local isValid = false
    for _, validPos in ipairs(validPositions) do
        if position == validPos then
            isValid = true
            break
        end
    end

    if not isValid then
        error("Icon position must be one of: 'left', 'right', 'top', 'bottom'")
    end

    self.iconPosition = position
    self:recalculateSize()
    self:_updateIconLayout()
    return self
end

---Sets padding
---@param padding integer
function UILib.Label:setIconPadding(padding)
    self.iconPadding = padding
    self:recalculateSize()
    self:_updateIconLayout()
    return self
end

function UILib.Label:_updateIconLayout()
    if not self.icon then return end

    self.icon:setSize(self.iconSize, self.iconSize)

    if self.iconPosition == "left" then
        self.icon:setPos(0, (self.h - self.iconSize) / 2)
    elseif self.iconPosition == "right" then
        self.icon:setPos(self.w - self.iconSize, (self.h - self.iconSize) / 2)
    elseif self.iconPosition == "top" then
        self.icon:setPos((self.w - self.iconSize) / 2, 0)
    elseif self.iconPosition == "bottom" then
        self.icon:setPos((self.w - self.iconSize) / 2, self.h - self.iconSize)
    end
end

function UILib.Label:setSize(w, h)
    UILib.Component.setSize(self, w, h)
    self:_updateIconLayout()
end

function UILib.Label:setPos(x, y)
    UILib.Component.setPos(self, x, y)
    self:_updateIconLayout()
end

function UILib.Label:_draw()
    local textX, textY = self.x, self.y

    if self.icon then
        self.icon:_draw()

        if self.iconPosition == "left" then
            textX = self.icon.x + self.iconSize + self.iconPadding
        elseif self.iconPosition == "right" then
            textX = self.x
        elseif self.iconPosition == "top" then
            textY = self.icon.y + self.iconSize + self.iconPadding
        elseif self.iconPosition == "bottom" then
            textY = self.y
        end
    end

    if self.font then render.setFont(self.font) end
    if self.textColor then render.setColor(self.textColor) end
    render.drawSimpleText(textX, textY, self.text)
end

function UILib.Label:recalculateSize()
    if self.font then render.setFont(self.font) else render.setFont(UILib.font) end
    local textWidth, textHeight = render.getTextSize(self.text)

    if self.icon then
        if self.iconPosition == "left" or self.iconPosition == "right" then
            self.w = textWidth + self.iconPadding + self.iconSize
            self.h = math.max(textHeight, self.iconSize)
        else
            self.w = math.max(textWidth, self.iconSize)
            self.h = textHeight + self.iconPadding + self.iconSize
        end
    else
        self.w = textWidth
        self.h = textHeight
    end
end

function UILib.Label:setText(text)
    self.text = text
    self:recalculateSize()
    if self.alignment then self:setAlignment(self.alignment) end
    self:_updateIconLayout()
end

---@param materialOrComponent string | UILib.Icon
---@param size integer | nil
---@param color Color | nil
---@param position string | nil
function UILib.Label:setIcon(materialOrComponent, size, color, position)
    if self.icon then
        self.icon:destroy()
        self.icon = nil
    end

    self.iconSize = size or 16
    self.iconColor = color

    if position then
        self:setIconPosition(position)
    end

    if type(materialOrComponent) == "string" then
        self.icon = UILib.Icon:new(self)
        self.icon:setSize(self.iconSize, self.iconSize)
        self.icon:setImage(materialOrComponent)
    else
        self.icon = materialOrComponent
        self.icon:setParent(self)
        self.icon:setSize(self.iconSize, self.iconSize)
    end

    if self.iconColor and self.icon.setColor then
        self.icon:setColor(self.iconColor)
    end

    self:recalculateSize()
    self:_updateIconLayout()
    return self
end
