---@class UILib.Icon: UILib.Component
UILib.Icon = class("Icon", UILib.Component)

function UILib.Icon:initialize(parent, screen)
    UILib.Component.initialize(self, 0, 0, 32, 32, false, parent, screen)
    
    self._atlasDrawable = nil
end


function UILib.Icon:setMaterial(path)
    if self._atlasDrawable then
        UILib.TextureAtlas:remove(self._atlasDrawable._id)
        self._atlasDrawable = nil
    end
    self._atlasDrawable = UILib.TextureAtlas:addMaterial(path, self, nil, self.w, self.h)
    return self
end

function UILib.Icon:setImage(path)
    if self._atlasDrawable then
        UILib.TextureAtlas:remove(self._atlasDrawable._id)
        self._atlasDrawable = nil
    end
    self._atlasDrawable = UILib.TextureAtlas:addImage(path, self, nil, self.w, self.h)
    return self
end

function UILib.Icon:setTextureURL(url)
    if self._atlasDrawable then
        UILib.TextureAtlas:remove(self._atlasDrawable._id)
        self._atlasDrawable = nil
    end
    self._atlasDrawable = UILib.TextureAtlas:addTextureURL(url, self, nil, self.w, self.h)
    return self
end


function UILib.Icon:_draw()
    if not self._atlasDrawable then return end
    self._atlasDrawable:_draw()
end

function UILib.Icon:destroy()
    if self._atlasDrawable then
        UILib.TextureAtlas:remove(self._atlasDrawable._id)
        self._atlasDrawable = nil
    end
    UILib.Component.destroy(self)
end
